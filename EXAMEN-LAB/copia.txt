#include <stdlib.h>
#include <time.h>


// esto va dentro del main
srand((int)time(NULL));



int option, max = -1;


do {
	// este es el mensaje para pedir cantidad de notas
	std::cout << "Ingresa la cantidad de notas"<< std::endl;
	std::cin >> max;
	
}while(max < 0 || max > 30);


int * notas = new int[max];
int totalA = 0, totalD = 0;
// generar notas 
for(int i = 0; i < max; i++){
   notas[i] = rand() % 7 + (13 - 7  + 1);
   
   if(this->notas[i] > 10) totalA++;
   else totalD++;
	
}

// imprimir notas

for(int i = 0; i < max; i++){
    std::cout <<  i << "=" << notas[i] << std::endl;
}


std::cout << "Total aprobados: " << totalA <<std::endl;
std::cout << "Total desaprobados: " << totalD  <<std::endl;
